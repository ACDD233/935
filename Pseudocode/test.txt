Class Tester (测试类):

    Function test_all_folds:
        // 循环所有 K-Fold
        for fold from 1 to N_splits:
            call test_single_fold(fold)
        
        // 保存 K-Fold 测试总结
        call save_summary_results()

    Function test_single_fold(fold_num):
        // 1. 加载此折 (fold) 训练好的模型
        model = load_model(fold_num)
        
        // 2. 获取此折的验证数据集
        validation_data = get_validation_data(fold_num)
        
        // 3. (关键) 根据配置 (white/field/mixed) 过滤数据
        filtered_data = filter_data_by_background(validation_data)
        
        // 4. 调用评估 API (model.val)
        metrics = model.val(filtered_data)
        
        // 5. 记录评估指标
        record_fold_metrics(metrics)
