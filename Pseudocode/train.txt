Class Trainer (训练类):

    Function train_all_folds:
        // 循环所有 K-Fold
        for fold from 1 to N_splits:
            call train_single_fold(fold)
        
        // 保存 K-Fold 总结
        call save_summary_results()

    Function train_single_fold(fold_num):
        // 1. 从 fold 文件夹中获取数据
        data = get_data_from_folder(fold_num)
        
        // 2. 加载 YOLO 模型
        model = load_yolo_model()
        
        // 3. 调用训练 API
        results = model.train(data, config_params)
        
        // 4. 保存模型和结果
        save_best_model(results.best_model)
        record_fold_metrics(results.metrics)
