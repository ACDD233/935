Class GradCAMVisualizer (可视化类):

    Function visualize_all_folds:
        // 循环所有 K-Fold
        for fold from 1 to N_splits:
            call visualize_fold(fold)

    Function visualize_fold(fold_num):
        // 1. 加载此折 (fold) 的模型
        model = load_model(fold_num)
        
        // 2. 从每个 (类别) folder 中取 N 张图片
        for each class_folder in validation_set:
            sample_images = get_random_images(class_folder, N_samples)
            
            for image in sample_images:
                // 3. 调用 API (生成热图)
                heatmap = call generate_gradcam_api(model, image)
                
                // 4. 获得 (并保存) 结果图片
                // (将原图、热图、叠加图合并保存)
                save_visualization(image, heatmap)
